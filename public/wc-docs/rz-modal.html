rz-modal


Syntax

Within HTML header: 
<script type="module" src="/assets/rz-modal.js"></script>
Within HTML body: 
<rz-modal id="rzmodal" attribute="value" > <rz-modal/>

default

<rz-modal
  id="rzmodal"
  title="Salute"
  body="Hello World 你好世界 Hola Mundo Привет мир Hallo Welt!"
  footer=""
  color="info"
>
<rz-modal/>

Attributes

attribute value description
title <title> title of the modal window
body <body> text within the modal window
footer <footer> optional, text of the modal window footer
color <info>|<red>|<yellow>|<green> optional, color of the modal window


Usage

Example usage

    <script>
      function showInfo() {
        const modal = document.getElementById("rzmodal");
        if (modal) {
          const dialog = modal.shadowRoot.querySelector("#rz_modal");
          dialog.showModal({ backdrop: true });
        }
      }
      function showGreen() {
        const modal = document.getElementById("rzmodal");
        modal.setAttribute("title", "I'm a green title");
        modal.setAttribute("body", "I'm the green body");
        modal.setAttribute("footer", "I'm a green footer");
        modal.setAttribute("color", "green");
        if (modal) {
          const dialog = modal.shadowRoot.querySelector("#rz_modal");
          dialog.showModal({ backdrop: true });
        }
      }
